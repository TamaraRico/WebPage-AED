Código

Para la implementación del algoritmo Radix Sort se utilizó Java, donde se implementó un código que incluye un menú con las opciones de LSD radix sort y MSD radix sort; así como una opción para números enteros y otro para palabras. El programa es el siguiente:

radixSort

public class radixSort {
    public static void radixSort(int arreglo[]) {
        int tam=arreglo.length;
        int max = getMax(arreglo, tam);

        for (int indice=1; max/indice>0; indice*=10)
            countingSort(arreglo, tam, indice);
    }

    public static void countingSort(int arreglo[], int tam, int indice) {
        int[] cuenta=new int[10];
        for (int valor:arreglo)
            cuenta[(valor/indice)%10]++;
        for (int i=1; i < 10; i++)
            cuenta[i]+=cuenta[i-1];
        int[] output = new int[tam];
        for (int i=tam-1; i >= 0; i--) {
            output[cuenta[(arreglo[i]/indice)%10]-1]=arreglo[i];
            cuenta[(arreglo[i]/indice) % 10]--;
        }
        for (int i=0; i < tam ; i++)
            arreglo[i] = output[i];
    }
    public static int getMax(int arreglo[], int tam) {
        int max=arreglo[0];
        for (int i=1; i < tam ; i++) {
            if (arreglo[i]>max)
                max = arreglo[i];
        }
    return max;
}

msdRadixSort

public class msdRadixSort {
    public static void msdradixSort(String[] arreglo){
        int max = getMax(arreglo);
        char menor='a', mayor='z';
        for(int i=max; i>=0; i--){
            countingSort(arreglo, i, menor, mayor);
        }
    }
    public static void countingSort(String[] arreglo, int indice, char menor, char mayor){
        int[] cuenta = new int[(mayor-menor)+2];
        String[] aux = new String[arreglo.length];
        Arrays.fill(cuenta,0);
        for(int i=0; i < arreglo.length; i++){
            int charIndice = (arreglo[i].length()-1 < indice) ? 0 : ((arreglo[i].charAt(indice)-menor)+1);
            cuenta[charIndice]++;
        }
        for(int i=1; i < cuenta.length; i++){
            cuenta[i]+=cuenta[i-1];
        }
        for(int i=arreglo.length-1; i>=0 ;i--){
            int charIndex = (arreglo[i].length()-1 < indice) ? 0 : (arreglo[i].charAt(indice) - menor)+1;
            aux[cuenta[charIndex]-1] = arreglo[i];
            cuenta[charIndex]--;
        }
        for(int i=0; i < aux.length; i++){
            arreglo[i] = aux[i];
        }
    }
    public static int getMax(String[] arreglo) {
        int maxIndice = 0;
        for(int i=0; i < arreglo.length; i++){
            if(arreglo[i].length()-1>maxIndice){
                maxIndice = arreglo[i].length()-1;
            }
        }
        return maxIndice;
    }
}
